  ```typescript
  function countFormulaAtoms(formulaText: string): number {
    // Sum element counts in a formula like 'C6H6' or 'C2H6O' or 'C3H7NO2'
    // Each element token is [A-Z][a-z]? optionally followed by digits
    let total = 0;
    for (const m of formulaText.matchAll(/[A-Z][a-z]?(\d*)/g)) {
      // skip hydrogen — formula layer highlights heavy atoms only
      // actually include all atoms that have canonical indices: heavy atoms
      // H is not a canonical heavy atom; canonical numbering is for heavy atoms
      total += m[1] ? parseInt(m[1], 10) : 1;
    }
    // Subtract hydrogen count (H atoms are not given canonical indices)
    // Re-parse: match non-H element tokens only
    total = 0;
    for (const m of formulaText.matchAll(/(?<![A-Za-z])([A-Z][a-z]?)(\d*)(?![a-z])/g)) {
      if (m[1] !== 'H') total += m[2] ? parseInt(m[2], 10) : 1;
    }
    return total;
  }

  function enrichLayers(layers: Layer[]): Layer[] {
    // Find the formula layer to get total heavy atom count for 'formula' enrichment
    const formulaLayer = layers.find(l => l.type === 'formula');
    const N = formulaLayer ? countFormulaAtoms(formulaLayer.text) : 0;

    return layers.map(layer => {
      switch (layer.type) {
        case 'formula':
          return { ...layer, atoms: Array.from({ length: N }, (_, i) => i + 1) };
        case 'c': {
          const bonds = parseConnectionBonds(layer.text);
          const atomSet = new Set<number>();
          bonds.forEach(([a, b]) => { atomSet.add(a); atomSet.add(b); });
          return { ...layer, atoms: [...atomSet].sort((a, b) => a - b), bonds };
        }
        case 'h': {
          const fixed = Object.keys(parseHydrogenAtoms(layer.text)).map(Number);
          const mobile = parseMobileHydrogens(layer.text);
          const all = [...new Set([...fixed, ...mobile])].sort((a, b) => a - b);
          return { ...layer, atoms: all };
        }
        case 't':
        case 'b':
          return { ...layer, atoms: parseStereoAtoms(layer.text) };
        default:
          return layer; // version, m, s, q, p, i — atoms: [], bonds: []
      }
    });
  }
  ```

  Finally add `parseInchi`:
  ```typescript
  export function parseInchi(s: string): Layer[] {
    const body = s.slice('InChI='.length);
    const parts = body.split('/');
    const layers: Layer[] = [];
    layers.push({ type: 'version', prefix: '', text: parts[0], atoms: [], bonds: [] });
    if (parts[1]) {
      layers.push({ type: 'formula', prefix: '', text: parts[1], atoms: [], bonds: [] });
    }
    for (let i = 2; i < parts.length; i++) {
      const p = parts[i];
      if (!p) continue;
      const prefix = p[0] as LayerType;
      layers.push({ type: prefix, prefix, text: p.slice(1), atoms: [], bonds: [] });
    }
    return enrichLayers(layers);
  }
  ```

  **Step 3 — Create `src/lib/parseAuxMapping.ts`:**

  ```typescript
  import { parseInchi } from './parseInchi';
  import type { Layer, AuxMap } from './parseInchi';

  export function parseAuxMapping(auxBody: string): AuxMap {
    // auxBody = the portion AFTER "AuxInfo=" e.g. "1/0/N:1,2,6,3,5,4/E:..."
    const parts = auxBody.split('/');
    const nPart = parts.find(p => p.startsWith('N:'));
    if (!nPart) return {};
    const values = nPart.slice(2).split(',');
    const map: AuxMap = {};
    values.forEach((v, i) => {
      const n = parseInt(v, 10);
      if (!isNaN(n)) map[i + 1] = n - 1; // canonical (1-based) → Ketcher index (0-based)
    });
    return map;
  }

  export function parseInchiWithAux(raw: string): {
    inchi: string;
    layers: Layer[];
    auxMap: AuxMap;
  } {
    // Split on the literal "\nAuxInfo=" separator (per D-11)
    // OPEN QUESTION: if Ketcher 3.12.0 on Windows uses \r\n, this may fail.
    // Empirical verification required: run app in browser, call window.ketcher.getInchi(true),
    // log raw.indexOf('\n') vs raw.indexOf('\r\n'). Update this if separator differs.
    const sep = '\nAuxInfo=';
    const idx = raw.indexOf(sep);
    if (idx === -1) {
      // No AuxInfo — plain InChI (shouldn't happen with getInchi(true) on non-empty canvas)
      return { inchi: raw, layers: parseInchi(raw), auxMap: {} };
    }
    const inchiStr = raw.slice(0, idx);
    const auxBody = raw.slice(idx + sep.length);
    return {
      inchi: inchiStr,
      layers: parseInchi(inchiStr),
      auxMap: parseAuxMapping(auxBody),
    };
  }
  ```
